#include<iostream>
using namespace std;
class Cube
{
private:
    int L;
    int W;
    int H;
public:
    int area() {
        return 2 * (L * W + L * H + W * H);
    }
    int volume() {
        return L * W * H;
    }
    int getL() {
        return L;
    }
    int getW() {
        return W;
    }
    int getH() {
        return H;
    }
    void setL(int l) {
        L = l;
    }
    void setW(int w) {
        W = w;
    }
    void setH(int h) {
        H = h;
    }
    bool isSame_1(Cube& c, Cube& c2) {
        if (c.H == c2.H && c.W == c.W && c.L == c2.L) {
            return true;
        }
        else {
            return false;
        }
    }
};
bool isSame(Cube& c, Cube& c2) {
    if (c.getH() == c2.getH() && c.getW() == c.getW() && c.getL() == c2.getL()) {
        return true;
    }
    else {
        return false;
    }
}
int main() 
{
    Cube c;
    c.setL(3);
    c.setW(4);
    c.setH(5);
    cout << "面积：" << c.area() << " 体积：" << c.volume() << endl;
    Cube c2;
    c2.setL(3);
    c2.setW(4);
    c2.setH(5);
    cout << "全局函数" << isSame(c, c2) << endl;
    cout << "类内函数" << c.isSame_1(c, c2) << endl;
    system("pause");
    return 0;
}

递增运算符重载

前置递增

后置递增

后置递增要返回值，因为temp是局部函数的类，函数运行完之后会自动释放掉

#include<iostream>
using namespace std;
class Person
{
    friend ostream& operator<< (ostream& cout, Person& p);
public:
    Person(int a, int b) {
        this->m_A = a;
        this->m_B = b;
    }
    Person& operator++()
    {
        this->m_A++;
        this->m_B++;
        return *this;
    }
    Person& operator++(int)
    {
        Person temp = *this;
        this->m_A++;
        this->m_B++;
        return temp;
    }
    Person& operator--()
    {
        this->m_A--;
        this->m_B--;
        return *this;
    }
    Person& operator--(int)
    {
        Person temp = *this;
        this->m_A--;
        this->m_B--;
        return temp;
    }
private:

    int m_A;
    int m_B;
};

ostream& operator<< (ostream& cout, Person& p)
{
    cout << p.m_A << " " << p.m_B;
    return cout;
}



void test()
{
    Person p(10, 10);
    cout << ++p << " hello world!" << endl;
    cout << --p << " hello world!" << endl;
    cout << p++ << " hello world!" << endl;
    cout << p-- << " hello world!" << endl;
}

int main()
{
    test();
    system("pause");
    return 0;
}

#include<iostream>
using namespace std;
class Person
{
    friend ostream& operator<< (ostream& cout, Person& p);
    friend bool operator== (Person &p1, Person &p2);
    friend bool operator!= (Person &p1, Person &p2);
public:
    Person(int a, int b) {
        m_A = new int(a);
        m_B = new int(b);
    }
    Person& operator= (Person& p) {
        if (m_A != NULL) {
            delete m_A;
            m_A = NULL;
        }
        if (m_B != NULL) {
            delete m_B;
            m_B = NULL;
        }
        m_A = new int(*p.m_A);
        m_B = new int(*p.m_B);
        return *this;
    }
    ~Person() {
        if (m_A != NULL) {
            delete m_A;
            m_A = NULL;
        }
        if (m_B != NULL) {
            delete m_B;
            m_B = NULL;
        }
    }
private:
    int* m_A;
    int* m_B;
};

bool operator== (Person &p1, Person &p2) {
    if (*p1.m_A == *p2.m_A && *p1.m_B == *p2.m_B) {
        return true;
    }
    else return false;
}

bool operator!= (Person &p1,Person &p2) {
    if (*p1.m_A == *p2.m_A && *p1.m_B == *p2.m_B) {
        return false;
    }
    else return true;
}

ostream& operator<< (ostream& cout, Person& p)
{
    cout << *p.m_A << " " << *p.m_B;
    return cout;
}



void test()
{
    Person p(10, 10);
    Person p1(20, 20);
    Person p2(30, 30);
    p2 = p1 = p;
    cout << p << " " << p1 << " " << p2 << " " << endl;
    if (p1 == p2 && p == p2) {
        cout <<"相等" << endl;
    }
    else {
        cout << "不相等" << endl;
    }
}

#include<iostream>
using namespace std;

class CPU
{
public:
    virtual void calculate() = 0;
};

class VideoCard
{
public:
    virtual void display() = 0;
};

class Memmory
{
public:
    virtual void storage() = 0;
};

class Computer
{
public:
    Computer(CPU* cpu, VideoCard* vc, Memmory* mem) {
        this->m_cpu = cpu;
        this->m_vc = vc;
        this->m_mem = mem;
    }
    ~Computer() {
        if (m_cpu != NULL) {
            delete m_cpu;
            m_cpu = NULL;
        }
        if (m_vc != NULL) {
            delete m_vc;
            m_vc = NULL;
        }
        if (m_mem != NULL) {
            delete m_mem;
            m_mem = NULL;
        }
    }
    void work() {
        m_cpu->calculate();
        m_vc->display();
        m_mem->storage();
    }
private:
    CPU* m_cpu;
    VideoCard* m_vc;
    Memmory* m_mem;
};

class intelCPU:public CPU
{
public:
    virtual void calculate() {
        cout << "intel的CPU开始计算了！" << endl;
    }
};

class intelVideoCard :public VideoCard
{
public:
    virtual void display() {
        cout << "intel的显卡开始显示了！" << endl;
    }
};

class intelMemmory :public Memmory
{
public:
    virtual void storage() {
        cout << "intel的存储条开始存储了！" << endl;
    }
};

class LevonoCPU :public CPU
{
public:
    virtual void calculate() {
        cout << "Levono的CPU开始计算了！" << endl;
    }
};

class LevonoVideoCard :public VideoCard
{
public:
    virtual void display() {
        cout << "Levono的显卡开始显示了！" << endl;
    }
};

class LevonoMemmory :public Memmory
{
public:
    virtual void storage() {
        cout << "Levono的存储条开始存储了！" << endl;
    }
};

void test()
{
    //组装第一台电脑
    Computer* computer1 = new Computer(new intelCPU, new intelVideoCard, new intelMemmory);
    computer1->work();
    delete computer1;
    cout << "-------------------------------" << endl;
    //组装第二台电脑
    Computer* computer2 = new Computer(new LevonoCPU, new LevonoVideoCard, new LevonoMemmory);
    computer2->work();
    delete computer2;
    cout << "-------------------------------" << endl;
    //组装第三台电脑
    Computer* computer3 = new Computer(new intelCPU, new LevonoVideoCard, new LevonoMemmory);
    computer3->work();
    delete computer3;
}

int main()
{
    test();
    system("pause");
    return 0;
}
