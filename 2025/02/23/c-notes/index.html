int isPrime(int i, int prime[],int count)
{
    int ret = 0;
    int j;
    for (j = 0; j < count; j++) {
        if (i % prime[j] == 0) {
            ret = 0;
            break;
        }

void minmax(int a[], int* min, int* max, int len)
{
    *min = a[0];
    *max = a[0];
    int i;
    for (i = 1; i < len; i++) {
        if (a[i] < *min) {
            *min = a[i];
        }
        if (a[i] > *max) {
            *max = a[i];
        }
    }
}
