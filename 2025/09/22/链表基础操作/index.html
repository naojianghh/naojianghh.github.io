链表基础操作

2025/9/22 算法

链表的定义及增删改查

1. 链表节点定义

单链表节点包含数据域和指针域（指向下一个节点）：

#include <iostream>
using namespace std;

// 节点结构体定义
struct Node {
    int data;       // 数据域
    Node* next;     // 指针域，指向下一个节点
    // 构造函数，方便初始化
    Node(int val) : data(val), next(nullptr) {}
};

2. 链表初始化

创建一个空链表（通常用头指针 / 哨兵节点表示）：

// 方法1：使用头指针（初始为nullptr）
Node* head = nullptr;

// 方法2：使用哨兵节点（简化操作，不存储实际数据）
Node* sentinel = new Node(0);  // 哨兵节点数据无意义

3. 插入操作

3.1 头插法（在链表头部插入）

// 在链表头部插入新节点
void insertAtHead(Node*& head, int val) {
    Node* newNode = new Node(val);  // 创建新节点
    newNode->next = head;           // 新节点指向原头节点
    head = newNode;                 // 更新头指针
}

// 示例
int main() {
    Node* head = nullptr;
    insertAtHead(head, 3);  // 链表：3
    insertAtHead(head, 2);  // 链表：2->3
    insertAtHead(head, 1);  // 链表：1->2->3
    return 0;
}

3.2 尾插法（在链表尾部插入）

// 在链表尾部插入新节点
void insertAtTail(Node*& head, int val) {
    Node* newNode = new Node(val);
    if (head == nullptr) {  // 空链表直接作为头节点
        head = newNode;
        return;
    }
    // 遍历到尾部
    Node* curr = head;
    while (curr->next != nullptr) {
        curr = curr->next;
    }
    curr->next = newNode;  // 尾部节点指向新节点
}

// 示例
int main() {
    Node* head = nullptr;
    insertAtTail(head, 1);  // 链表：1
    insertAtTail(head, 2);  // 链表：1->2
    insertAtTail(head, 3);  // 链表：1->2->3
    return 0;
}

3.3 中间插入（在指定位置插入）

// 在第k个节点后插入新节点（k从1开始计数）
void insertAtPosition(Node*& head, int k, int val) {
    if (k <= 0) return;  // 位置无效
    // 若插入到头部（k=0的特殊情况）
    if (k == 1 && head == nullptr) {
        insertAtHead(head, val);
        return;
    }
    Node* curr = head;
    // 遍历到第k-1个节点
    for (int i = 1; i < k - 1 && curr != nullptr; ++i) {
        curr = curr->next;
    }
    if (curr == nullptr) return;  // 位置超出链表长度
    Node* newNode = new Node(val);
    newNode->next = curr->next;   // 新节点指向原k位置节点
    curr->next = newNode;         // 原k-1节点指向新节点
}

// 示例
int main() {
    Node* head = nullptr;
    insertAtTail(head, 1);
    insertAtTail(head, 3);
    insertAtPosition(head, 2, 2);  // 在第1个节点后插入2，链表：1->2->3
    return 0;
}

4. 删除操作

4.1 删除头节点

void deleteHead(Node*& head) {
    if (head == nullptr) return;  // 空链表
    Node* temp = head;    // 暂存头节点
    head = head->next;    // 更新头指针
    delete temp;          // 释放内存
}

// 示例
int main() {
    Node* head = nullptr;
    insertAtTail(head, 1);
    insertAtTail(head, 2);
    deleteHead(head);  // 链表变为：2
    return 0;
}

4.2 删除尾节点

void deleteTail(Node*& head) {
    if (head == nullptr) return;  // 空链表
    if (head->next == nullptr) {  // 只有一个节点
        delete head;
        head = nullptr;
        return;
    }
    Node* curr = head;
    // 遍历到倒数第二个节点
    while (curr->next->next != nullptr) {
        curr = curr->next;
    }
    delete curr->next;    // 释放尾节点
    curr->next = nullptr; // 倒数第二个节点指向nullptr
}

// 示例
int main() {
    Node* head = nullptr;
    insertAtTail(head, 1);
    insertAtTail(head, 2);
    deleteTail(head);  // 链表变为：1
    return 0;
}

4.3 删除指定值的节点

void deleteByValue(Node*& head, int val) {
    if (head == nullptr) return;
    // 若头节点是目标值
    if (head->data == val) {
        deleteHead(head);
        return;
    }
    Node* curr = head;
    // 找到目标节点的前一个节点
    while (curr->next != nullptr && curr->next->data != val) {
        curr = curr->next;
    }
    if (curr->next == nullptr) return;  // 未找到目标值
    Node* temp = curr->next;
    curr->next = curr->next->next;  // 跳过目标节点
    delete temp;                    // 释放内存
}

// 示例
int main() {
    Node* head = nullptr;
    insertAtTail(head, 1);
    insertAtTail(head, 2);
    insertAtTail(head, 3);
    deleteByValue(head, 2);  // 链表变为：1->3
    return 0;
}

5. 遍历操作

// 遍历并打印链表所有元素
void traverse(Node* head) {
    Node* curr = head;
    while (curr != nullptr) {
        cout << curr->data << "->";
        curr = curr->next;
    }
    cout << "nullptr" << endl;
}

// 示例
int main() {
    Node* head = nullptr;
    insertAtTail(head, 1);
    insertAtTail(head, 2);
    insertAtTail(head, 3);
    traverse(head);  // 输出：1->2->3->nullptr
    return 0;
}

6. 查找操作

// 查找值为val的节点，返回是否存在
bool find(Node* head, int val) {
    Node* curr = head;
    while (curr != nullptr) {
        if (curr->data == val) {
            return true;
        }
        curr = curr->next;
    }
    return false;
}

// 示例
int main() {
    Node* head = nullptr;
    insertAtTail(head, 1);
    insertAtTail(head, 2);
    cout << find(head, 2) << endl;  // 输出：1（存在）
    cout << find(head, 3) << endl;  // 输出：0（不存在）
    return 0;
}

7. 销毁链表（释放内存）

void destroy(Node*& head) {
    while (head != nullptr) {
        Node* temp = head;
        head = head->next;
        delete temp;  // 逐个释放节点
    }
}

// 示例
int main() {
    Node* head = nullptr;
    insertAtTail(head, 1);
    insertAtTail(head, 2);
    destroy(head);  // 释放所有节点，head变为nullptr
    return 0;
}

总结

单链表的核心操作围绕指针的修改展开，关键是要：

  1. 避免空指针访问（操作前判断节点是否为nullptr）
  2. 正确维护指针关系（插入 / 删除时确保链表不中断）
  3. 及时释放动态分配的内存（防止内存泄漏）
