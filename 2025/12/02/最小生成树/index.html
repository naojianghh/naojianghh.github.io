最小生成树

2025/12/2 算法 c++ 图

最小生成树是图论的核心应用之一，核心结论是：在连通无向加权图中，找到一棵包含所有顶点、边权总和最小且无回路的子图。

一、核心定义与前提

  • 适用场景：连通无向图（任意两点间有路径）、边带非负权重（避免负权导致无最优解）。
  • 核心要求：生成树需满足两个条件 —— 包含图中所有顶点，边的数量为 “顶点数 - 1”（无回路）。
  • 核心目标：所有满足条件的生成树中，边权之和最小。

二、两种经典算法（重点）

1. Prim 算法（加点法）

  • 核心思路：从任意顶点出发，逐步加入 “与当前生成树距离最近的顶点”，直到包含所有顶点。
  • 关键步骤：
    1. 初始化：选一个起点，标记为已访问，记录各顶点到生成树的最小距离。
    2. 迭代：每次选未访问顶点中距离最小的，加入生成树，更新剩余顶点的距离（取 “原距离” 和 “新边权” 的最小值）。
    3. 终止：所有顶点都被访问，生成最小生成树。
  • 适用场景：顶点数少、边数多的稠密图（时间复杂度 O (n²)，n 为顶点数）。

2. Kruskal 算法（加边法）

  • 核心思路：将所有边按权值从小到大排序，逐步加入边，避免形成回路，直到边数为 “顶点数 - 1”。
  • 关键步骤：
    1. 排序：所有边按权值升序排列。
    2. 选边：依次选边，若边的两个顶点不在同一连通分量（用并查集判断），则加入生成树。
    3. 终止：加入的边数达到 “顶点数 - 1”，停止选边。
  • 适用场景：顶点数多、边数少的稀疏图（时间复杂度 O (m log m)，m 为边数）。

三、核心性质

  • 唯一性：若图中所有边权都不相同，最小生成树唯一；若有相同边权，可能存在多棵最小生成树。
  • 切割性质：任意分割图为两部分，连接两部分的最小权边一定在最小生成树中。
  • 回路性质：任意回路中，权值最大的边一定不在最小生成树中。

四、代码实现（c++）

Prim算法

#include <iostream>
#include <vector>
#include <queue>
#include <climits> // 用于INT_MAX
using namespace std;

// Prim算法：返回最小生成树的总权值
// 参数：n（顶点数），graph（邻接表，格式：graph[u] = {{v1, w1}, {v2, w2}, ...}）
int prim(int n, const vector>>& graph) {
    int minTotalWeight = 0;  // 最小生成树总权值
    vector dist(n, INT_MAX);  // 各顶点到生成树的最小距离
    vector visited(n, false);  // 标记是否已加入生成树
    // 优先队列：存储（距离，顶点），按距离从小到大排序（最小堆）
    priority_queue, vector>, greater>> pq;

    // 步骤1：从顶点0开始初始化
    dist[0] = 0;
    pq.emplace(0, 0);

    // 步骤2：迭代选择距离最小的顶点
    while (!pq.empty()) {
        auto [currentDist, u] = pq.top();
        pq.pop();

        // 若顶点已访问，跳过（避免重复处理）
        if (visited[u]) continue;

        // 步骤3：加入生成树，累计权值
        visited[u] = true;
        minTotalWeight += currentDist;

        // 步骤4：更新邻接顶点的距离
        for (auto [v, w] : graph[u]) {
            if (!visited[v] && w < dist[v]) {
                dist[v] = w;
                pq.emplace(dist[v], v);
            }
        }
    }

    return minTotalWeight;
}

int main() {
    // 测试用例：5个顶点（0-4），边如下（u, v, weight）
    // 0-1(2), 0-3(6), 1-2(3), 1-3(8), 1-4(5), 2-4(7), 3-4(9)
    int n = 5;
    vector>> graph(n);
    graph[0] = {{1, 2}, {3, 6}};
    graph[1] = {{0, 2}, {2, 3}, {3, 8}, {4, 5}};
    graph[2] = {{1, 3}, {4, 7}};
    graph[3] = {{0, 6}, {1, 8}, {4, 9}};
    graph[4] = {{1, 5}, {2, 7}, {3, 9}};

    int result = prim(n, graph);
    cout << "Prim算法最小生成树总权值：" << result << endl; // 预期输出：16

    return 0;
}

Kruskal算法

#include <iostream>
#include <vector>
#include <algorithm> // 用于sort
using namespace std;

// 边的结构体：存储两个顶点和边权
struct Edge {
    int u, v;     // 边的两个顶点
    int weight;   // 边权
    // 重载<运算符，用于按边权从小到大排序
    bool operator<(const Edge& other) const {
        return weight < other.weight;
    }
};

// 并查集：用于判断顶点是否连通（路径压缩+按秩合并）
class UnionFind {
private:
    vector<int> parent; // 父节点数组
    vector<int> rank;   // 秩（用于按秩合并）
public:
    UnionFind(int n) {
        parent.resize(n);
        rank.resize(n, 0);
        for (int i = 0; i < n; i++) {
            parent[i] = i; // 初始时每个节点的父节点是自己
        }
    }

    // 查找根节点（路径压缩）
    int find(int x) {
        if (parent[x] != x) {
            parent[x] = find(parent[x]); // 递归压缩路径
        }
        return parent[x];
    }

    // 合并两个集合（按秩合并）
    bool unite(int x, int y) {
        int rootX = find(x);
        int rootY = find(y);
        if (rootX == rootY) {
            return false; // 已连通，无需合并
        }
        // 按秩合并：将秩小的树合并到秩大的树下
        if (rank[rootX] < rank[rootY]) {
            parent[rootX] = rootY;
        } else {
            parent[rootY] = rootX;
            if (rank[rootX] == rank[rootY]) {
                rank[rootX]++; // 秩相等时，合并后秩+1
            }
        }
        return true; // 合并成功
    }
};

// Kruskal算法：返回最小生成树的总权值
// 参数：n（顶点数），edges（所有边的列表）
int kruskal(int n, vector<Edge>& edges) {
    int minTotalWeight = 0; // 最小生成树总权值
    int edgeCount = 0;      // 已选边数
    UnionFind uf(n);        // 初始化并查集

    // 步骤1：按边权从小到大排序
    sort(edges.begin(), edges.end());

    // 步骤2：遍历边，选n-1条不形成回路的边
    for (auto& edge : edges) {
        int u = edge.u;
        int v = edge.v;
        int w = edge.weight;

        // 若u和v不在同一连通分量，合并并加入生成树
        if (uf.unite(u, v)) {
            minTotalWeight += w;
            edgeCount++;
            if (edgeCount == n - 1) {
                break; // 已选够n-1条边，生成树完成
            }
        }
    }

    return minTotalWeight;
}

int main() {
    // 测试用例：5个顶点（0-4），边如下（u, v, weight）
    // 0-1(2), 0-3(6), 1-2(3), 1-3(8), 1-4(5), 2-4(7), 3-4(9)
    int n = 5;
    vector<Edge> edges = {
        {0, 1, 2}, {0, 3, 6}, {1, 2, 3}, {1, 3, 8}, {1, 4, 5},
        {2, 4, 7}, {3, 4, 9}
    };

    int result = kruskal(n, edges);
    cout << "Kruskal算法最小生成树总权值：" << result << endl; // 预期输出：16

    return 0;
}
