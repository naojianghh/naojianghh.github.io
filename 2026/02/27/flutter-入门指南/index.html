Flutter 基础组件

Flutter 提供了丰富的基础组件，用于构建用户界面：

1. 文本组件

  • Text：显示文本内容
  • RichText：显示富文本
  • TextField：文本输入框

2. 布局组件

  • Container：容器组件，可设置宽高、边距、背景等
  • Row：水平布局
  • Column：垂直布局
  • Stack：堆叠布局
  • ListView：列表布局
  • GridView：网格布局

3. 交互组件

  • Button：按钮
  • Icon：图标
  • Image：图片
  • Checkbox：复选框
  • Radio：单选按钮
  • Switch：开关

状态管理

Flutter 有多种状态管理方案：

1. setState

最基本的状态管理方式，适用于简单的状态管理。

class Counter extends StatefulWidget {
  @override
  _CounterState createState() => _CounterState();
}

class _CounterState extends State<Counter> {
  int _counter = 0;

  void _increment() {
    setState(() {
      _counter++;
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('Counter')),
      body: Center(
        child: Text('Count: $_counter'),
      ),
      floatingActionButton: FloatingActionButton(
        onPressed: _increment,
        child: Icon(Icons.add),
      ),
    );
  }
}

2. Provider

官方推荐的状态管理方案，适用于中型应用。

Provider 基本使用

  1. 添加依赖
dependencies:
  provider: ^6.0.0
  1. 创建数据模型
class CounterModel extends ChangeNotifier {
  int _count = 0;
  int get count => _count;

  void increment() {
    _count++;
    notifyListeners(); // 通知监听器状态发生变化
  }

  void decrement() {
    _count--;
    notifyListeners();
  }
}
  1. 在应用中提供数据
void main() {
  runApp(
    ChangeNotifierProvider(
      create: (context) => CounterModel(),
      child: MyApp(),
    ),
  );
}
  1. 在组件中使用数据
class CounterWidget extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Consumer<CounterModel>(
      builder: (context, counter, child) {
        return Column(
          children: [
            Text('Count: {counter.count}'),
            ElevatedButton(
              onPressed: counter.increment,
              child: Text('Increment'),
            ),
            ElevatedButton(
              onPressed: counter.decrement,
              child: Text('Decrement'),
            ),
          ],
        );
      },
    );
  }
}
  1. 多个 Provider 的使用
void main() {
  runApp(
    MultiProvider(
      providers: [
        ChangeNotifierProvider(create: (context) => CounterModel()),
        ChangeNotifierProvider(create: (context) => UserModel()),
      ],
      child: MyApp(),
    ),
  );
}

4. Riverpod

Provider 的改进版，提供了更灵活的状态管理方式。

Dio 请求

Dio 是 Flutter 中常用的网络请求库，支持 RESTful API、文件上传下载等功能。

1. 安装 Dio

在 pubspec.yaml 文件中添加依赖：

dependencies:
  dio: ^5.0.0

2. 基本使用

import 'package:dio/dio.dart';

void fetchData() async {
  try {
    var dio = Dio();
    var response = await dio.get('https://api.example.com/data');
    print(response.data);
  } catch (e) {
    print(e);
  }
}

3. 高级用法

  • 拦截器：用于添加请求头、处理认证等
  • 超时设置：设置请求超时时间
  • 取消请求：支持取消正在进行的请求
  • 表单提交：支持提交表单数据
  • 文件上传下载：支持上传和下载文件

Dio 封装使用

为了方便使用，我们可以对 Dio 进行封装，创建一个网络请求工具类。

1. 创建网络请求工具类

import 'package:dio/dio.dart';

class HttpUtil {
  static final HttpUtil _instance = HttpUtil._internal();
  factory HttpUtil() => _instance;

  Dio? dio;

  HttpUtil._internal() {
    BaseOptions options = BaseOptions(
      baseUrl: 'https://api.example.com', // 基础 URL
      connectTimeout: 10000, // 连接超时时间
      receiveTimeout: 10000, // 接收超时时间
      headers: {
        'Content-Type': 'application/json',
      },
    );

    dio = Dio(options);

    // 添加请求拦截器
    dio?.interceptors.add(InterceptorsWrapper(
      onRequest: (options, handler) {
        // 可以在这里添加 token 等认证信息
        // options.headers['Authorization'] = 'Bearer token';
        return handler.next(options);
      },
      onResponse: (response, handler) {
        // 对响应数据进行处理
        return handler.next(response);
      },
      onError: (DioError e, handler) {
        // 对错误进行处理
        print('请求错误: ${e.message}');
        return handler.next(e);
      },
    ));
  }

  // GET 请求
  Future<dynamic> get(String url, {Map<String, dynamic>? queryParameters}) async {
    try {
      Response response = await dio!.get(url, queryParameters: queryParameters);
      return response.data;
    } catch (e) {
      print('GET 请求错误: $e');
      rethrow;
    }
  }

  // POST 请求
  Future<dynamic> post(String url, {dynamic data}) async {
    try {
      Response response = await dio!.post(url, data: data);
      return response.data;
    } catch (e) {
      print('POST 请求错误: $e');
      rethrow;
    }
  }

  // PUT 请求
  Future<dynamic> put(String url, {dynamic data}) async {
    try {
      Response response = await dio!.put(url, data: data);
      return response.data;
    } catch (e) {
      print('PUT 请求错误: $e');
      rethrow;
    }
  }

  // DELETE 请求
  Future<dynamic> delete(String url) async {
    try {
      Response response = await dio!.delete(url);
      return response.data;
    } catch (e) {
      print('DELETE 请求错误: $e');
      rethrow;
    }
  }

  // 上传文件
  Future<dynamic> uploadFile(String url, String filePath) async {
    try {
      FormData formData = FormData.fromMap({
        'file': await MultipartFile.fromFile(filePath),
      });
      Response response = await dio!.post(url, data: formData);
      return response.data;
    } catch (e) {
      print('上传文件错误: $e');
      rethrow;
    }
  }

  // 下载文件
  Future<dynamic> downloadFile(String url, String savePath) async {
    try {
      Response response = await dio!.download(url, savePath);
      return response.data;
    } catch (e) {
      print('下载文件错误: $e');
      rethrow;
    }
  }
}

2. 使用封装的网络请求工具

// 初始化网络请求工具
final httpUtil = HttpUtil();

// GET 请求
void fetchData() async {
  try {
    var response = await httpUtil.get('/api/data', queryParameters: {'page': 1, 'size': 10});
    print(response);
  } catch (e) {
    print('请求失败: $e');
  }
}

// POST 请求
void submitData() async {
  try {
    var response = await httpUtil.post('/api/login', data: {
      'username': 'user',
      'password': 'pass',
    });
    print(response);
  } catch (e) {
    print('请求失败: $e');
  }
}

// 上传文件
void uploadImage() async {
  try {
    var response = await httpUtil.uploadFile('/api/upload', '/path/to/image.jpg');
    print(response);
  } catch (e) {
    print('上传失败: $e');
  }
}

// 下载文件
void downloadImage() async {
  try {
    var response = await httpUtil.downloadFile('https://example.com/image.jpg', '/path/to/save/image.jpg');
    print('下载成功');
  } catch (e) {
    print('下载失败: $e');
  }
}

3. 错误处理

在实际应用中，我们可以进一步完善错误处理，例如：

// 错误处理工具类
class ErrorUtil {
  static String handleError(dynamic error) {
    if (error is DioError) {
      switch (error.type) {
        case DioErrorType.connectTimeout:
          return '连接超时，请检查网络';
        case DioErrorType.sendTimeout:
          return '发送超时，请稍后重试';
        case DioErrorType.receiveTimeout:
          return '接收超时，请稍后重试';
        case DioErrorType.response:
          return '服务器错误，状态码: ${error.response?.statusCode}';
        case DioErrorType.cancel:
          return '请求已取消';
        case DioErrorType.other:
          return '网络错误，请检查网络连接';
        default:
          return '未知错误';
      }
    } else {
      return error.toString();
    }
  }
}

// 使用错误处理
void fetchData() async {
  try {
    var response = await httpUtil.get('/api/data');
    print(response);
  } catch (e) {
    String errorMessage = ErrorUtil.handleError(e);
    print('请求失败: $errorMessage');
    // 显示错误信息给用户
  }
}
