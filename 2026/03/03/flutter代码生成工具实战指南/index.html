Flutter 代码生成工具实战指南

2026/3/3 移动端开发 flutter 工具

json_serializable，riverpod_generator，freezed

什么是代码生成？

在 Flutter 开发中，我们经常需要编写大量重复性的样板代码，比如：

  • JSON 序列化/反序列化
  • equals()、hashCode()、toString() 方法
  • 不可变数据类
  • Provider 状态管理

代码生成工具能够根据我们的定义自动生成这些代码，让我们专注于业务逻辑。

为什么需要代码生成？

手动编写的问题

// 手动编写的数据类
class User {
  final String name;
  final int age;
  final String email;

  User({
    required this.name,
    required this.age,
    required this.email,
  });

  // 需要手动实现大量方法
  @override
  bool operator ==(Object other) {
    if (identical(this, other)) return true;
    return other is User &&
        other.name == name &&
        other.age == age &&
        other.email == email;
  }

  @override
  int get hashCode => name.hashCode ^ age.hashCode ^ email.hashCode;

  @override
  String toString() => 'User(name: $name, age: $age, email: $email)';

  factory User.fromJson(Map<String, dynamic> json) {
    return User(
      name: json['name'] as String,
      age: json['age'] as int,
      email: json['email'] as String,
    );
  }

  Map<String, dynamic> toJson() {
    return {
      'name': name,
      'age': age,
      'email': email,
    };
  }

  User copyWith({
    String? name,
    int? age,
    String? email,
  }) {
    return User(
      name: name ?? this.name,
      age: age ?? this.age,
      email: email ?? this.email,
    );
  }
}

使用代码生成后，只需几行代码就能实现相同的功能！

json_serializable 使用指南

适用场景

  • API 接口返回数据的解析
  • 本地数据存储（SharedPreferences、SQLite）
  • 数据模型的序列化/反序列化

安装依赖

dependencies:
  json_annotation: ^4.8.0

dev_dependencies:
  build_runner: ^2.4.0
  json_serializable: ^6.7.0

基本使用

1. 定义数据模型

import 'package:json_annotation/json_annotation.dart';

part 'user.g.dart';

@JsonSerializable()
class User {
  final String name;
  final int age;
  final String? email;
  final List<String> tags;

  User({
    required this.name,
    required this.age,
    this.email,
    this.tags = const [],
  });

  // 自动生成 fromJson 和 toJson
  factory User.fromJson(Map<String, dynamic> json) => _$UserFromJson(json);
  Map<String, dynamic> toJson() => _$UserToJson(this);
}

2. 嵌套对象

@JsonSerializable()
class Address {
  final String city;
  final String street;

  Address({required this.city, required this.street});

  factory Address.fromJson(Map<String, dynamic> json) => _$AddressFromJson(json);
  Map<String, dynamic> toJson() => _$AddressToJson(this);
}

@JsonSerializable()
class User {
  final String name;
  final Address address;

  User({required this.name, required this.address});

  factory User.fromJson(Map<String, dynamic> json) => _$UserFromJson(json);
  Map<String, dynamic> toJson() => _$UserToJson(this);
}

3. 自定义字段名

@JsonSerializable()
class User {
  @JsonKey(name: 'user_name')
  final String name;

  @JsonKey(name: 'user_age')
  final int age;

  @JsonKey(includeFromJson: false, includeToJson: false)
  final String? tempData;

  User({required this.name, required this.age, this.tempData});

  factory User.fromJson(Map<String, dynamic> json) => _$UserFromJson(json);
  Map<String, dynamic> toJson() => _$UserToJson(this);
}

4. 生成代码

# 生成代码
dart run build_runner build

# 监听文件变化自动重新生成
dart run build_runner watch

# 删除生成的文件
dart run build_runner clean

完整示例

// user.g.dart 自动生成内容
// GENERATED CODE - DO NOT MODIFY BY HAND

part of 'user.dart';

User _$UserFromJson(Map<String, dynamic> json) => User(
      name: json['name'] as String,
      age: json['age'] as int,
      email: json['email'] as String?,
      tags: (json['tags'] as List<dynamic>?)
          ?.map((e) => e as String)
          .toList() ?? const [],
    );

Map<String, dynamic> _$UserToJson(User instance) => <String, dynamic>{
      'name': instance.name,
      'age': instance.age,
      'email': instance.email,
      'tags': instance.tags,
    };

使用示例

void main() {
  // JSON 转对象
  final json = {'name': '张三', 'age': 25, 'email': 'zhangsan@example.com'};
  final user = User.fromJson(json);

  // 对象转 JSON
  final jsonString = user.toJson();
  print(jsonString);
}

freezed 使用指南

适用场景

  • 需要不可变数据类的场景
  • 需要完整的 equals、hashCode、toString 实现
  • 需要 copyWith 功能
  • 需要 union/sealed class 功能
  • 需要数据类之间的转换

安装依赖

dependencies:
  freezed_annotation: ^2.4.0
  json_annotation: ^4.8.0

dev_dependencies:
  build_runner: ^2.4.0
  freezed: ^2.4.0
  json_serializable: ^6.7.0

基本使用

1. 定义不可变数据类

import 'package:freezed_annotation/freezed_annotation.dart';

part 'user.freezed.dart';
part 'user.g.dart';

@freezed
class User with _$User {
  const factory User({
    required String name,
    required int age,
    String? email,
  }) = _User;
}

2. 使用 const 构造

void main() {
  // 创建不可变对象
  const user = User(name: '张三', age: 25);

  // 使用 copyWith
  final newUser = user.copyWith(age: 26);

  // 使用 == 和 hashCode
  print(user == newUser); // false

  // 使用 toString
  print(user.toString());
  // User(name: 张三, age: 25)
}

3. 工厂构造器

@freezed
class User with _$User {
  const factory User({
    required String name,
    required int age,
  }) = _User;

  // 工厂构造器
  factory User.fromJson(Map<String, dynamic> json) => _$UserFromJson(json);
}

4. Union/Sealed Class

@freezed
class Result with _$Result {
  const factory Result.success(String data) = Success;
  const factory Result.error(String message) = Error;
  const factory Result.loading() = Loading;
}

// 使用
void handleResult(Result result) {
  switch (result) {
    case Success():
      print('成功: ${result.data}');
    case Error():
      print('错误: ${result.message}');
    case Loading():
      print('加载中...');
  }
}

5. 复杂嵌套

@freezed
class Address with _$Address {
  const factory Address({
    required String city,
    required String street,
  }) = _Address;
}

@freezed
class User with _$User {
  const factory User({
    required String name,
    required Address address,
    @Default([]) List<String> tags,
  }) = _User;
}

riverpod_generator 使用指南

适用场景

  • 状态管理
  • 依赖注入
  • 异步数据管理
  • Provider 生命周期管理

安装依赖

dependencies:
  flutter_riverpod: ^2.4.0
  riverpod_annotation: ^2.3.0

dev_dependencies:
  build_runner: ^2.4.0
  riverpod_generator: ^2.3.0

基本使用

1. 定义 Provider

import 'package:riverpod_annotation/riverpod_annotation.dart';

part 'providers.g.dart';

// 简单的值 Provider
@riverpod
String greeting(GreetingRef ref) {
  return 'Hello, World!';
}

// 带参数的方法 Provider
@riverpod
int add(AddRef ref, int a, int b) {
  return a + b;
}

// 状态 Provider
@riverpod
class Counter extends _$Counter {
  @override
  int build() => 0;

  void increment() => state++;

  void decrement() => state--;
}

2. 异步 Provider

@riverpod
Future<String> fetchData(FetchDataRef ref) async {
  await Future.delayed(const Duration(seconds: 1));
  return '数据加载完成';
}

// 带错误处理的异步 Provider
@riverpod
class UserInfo extends _$UserInfo {
  @override
  FutureOr<String> build() async {
    try {
      await Future.delayed(const Duration(seconds: 1));
      return '用户信息';
    } catch (e) {
      return '加载失败: $e';
    }
  }
}

3. 依赖其他 Provider

@riverpod
int doubleValue(DoubleValueRef ref) {
  final counter = ref.watch(counterProvider);
  return counter * 2;
}

@riverpod
class User extends _$User {
  @override
  FutureOr<String> build() async {
    // 依赖其他 Provider
    final config = ref.watch(configProvider);
    return '用户: $config';
  }
}

4. 家族 Provider（带参数）

@riverpod
int multiplyBy(MultiplyByRef ref, int value) {
  final multiplier = ref.watch(multiplierProvider);
  return value * multiplier;
}

// 使用
final result = ref.watch(multiplyByProvider(5)); // 5 * multiplier

5. 生成的代码示例

// providers.g.dart 自动生成

// Provider
String greeting(GreetingRef ref) {
  return 'Hello, World!';
}

// StateNotifier Provider
class Counter extends _$Counter {
  @override
  int build() => 0;

  void increment() => state++;

  void decrement() => state--;
}

final counterProvider = AutoDisposeNotifierProvider<Counter, int>(Counter.new);

使用 Widget

import 'package:flutter_riverpod/flutter_riverpod.dart';

void main() {
  runApp(
    const ProviderScope(
      child: MyApp(),
    ),
  );
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    // 监听 Provider
    final count = ref.watch(counterProvider);

    return Scaffold(
      body: Center(child: Text('Count: $count')),
      floatingActionButton: FloatingActionButton(
        // 调用方法
        onPressed: () => ref.read(counterProvider.notifier).increment(),
        child: const Icon(Icons.add),
      ),
    );
  }
}

// user.dart
import 'package:freezed_annotation/freezed_annotation.dart';
import 'package:riverpod_annotation/riverpod_annotation.dart';
import 'package:json_annotation/json_annotation.dart';

part 'user.freezed.dart';
part 'user.g.dart';
part 'user.riverpod.dart';

@freezed
class User with _$User {
  const factory User({
    required String name,
    required int age,
    @Default([]) List<String> tags,
  }) = _User;

  factory User.fromJson(Map<String, dynamic> json) => _$UserFromJson(json);
}

// Riverpod Provider
@riverpod
class UserNotifier extends _$UserNotifier {
  @override
  FutureOr<User> build() async {
    // 模拟加载用户数据
    await Future.delayed(const Duration(seconds: 1));
    return const User(name: '张三', age: 25, tags: ['开发者', '程序员']);
  }

  void updateName(String name) {
    state = AsyncValue.data(
      state.value?.copyWith(name: name) ?? User(name: name, age: 0),
    );
  }
}

常见问题与解决方案

1. 代码生成失败

# 清理并重新生成
dart run build_runner clean
dart run build_runner build

2. import 路径错误

确保正确导入生成的文件：

// 正确：使用 part 导入
part 'user.g.dart';

// 错误：不要手动 import
// import 'user.g.dart';  // 不要这样做！

3. 类型不匹配

检查注解使用是否正确：

// 错误
@JsonKey(name: 'user_age')
final int age;

// 正确
@JsonKey(name: 'user_age')
final int userAge;

4. build_runner 占用资源过多

使用 --delete-conflicting-outputs 选项：

dart run build_runner build --delete-conflicting-outputs

总结

各工具适用场景

工具 适用场景
json_serializable API 数据解析、本地存储
freezed 不可变数据类、Union 类型
riverpod_generator 状态管理、依赖注入

最佳实践

  1. 小型项目：使用 json_serializable 即可
  2. 中型项目：json_serializable + freezed
  3. 大型项目：json_serializable + freezed + riverpod_generator

代码生成工具能够显著提高开发效率，减少样板代码，让开发者专注于业务逻辑的实现！
