Flutter EventBus 使用指南

2026/3/3 移动端开发 flutter

EventBus 是一个发布/订阅模式的事件总线库，它允许 Flutter 应用中的不同组件之间进行松耦合的通信。通过 EventBus，我们可以实现跨页面、跨组件的事件传递，而无需直接依赖对方的实例。

什么是 EventBus？

EventBus 的核心概念

  • Event（事件）：用户自定义的事件类，用于传递数据
  • Publisher（发布者）：发送事件的一方
  • Subscriber（订阅者）：接收并处理事件的一方
  • EventBus（事件总线）：管理事件发布和订阅的核心类

为什么需要 EventBus？

在 Flutter 开发中，我们通常使用以下方式进行组件间通信：

1. 构造函数传参

适用于父子组件通信，但会导致代码耦合度高。

2. Provider/Bloc

适用于状态管理，但学习曲线较陡峭。

3. EventBus

适用于简单的跨组件通信，代码简洁、易于理解。

EventBus 的优势：

  • 简单易用，API 直观
  • 松耦合，发布者和订阅者无需直接依赖
  • 支持任意类型的事件
  • 可以灵活地控制事件的作用域

EventBus 的基本使用

1. 添加依赖

在 pubspec.yaml 文件中添加 event_bus 依赖：

dependencies:
  event_bus: ^2.0.0

然后执行：

flutter pub get

2. 创建事件类

首先，我们需要定义我们的事件类：

class UserLoginEvent {
  final String username;
  final String token;

  UserLoginEvent(this.username, this.token);
}

class UserLogoutEvent {}

class MessageReceivedEvent {
  final String message;
  final DateTime timestamp;

  MessageReceivedEvent(this.message, this.timestamp);
}

3. 创建 EventBus 实例

可以在应用入口处创建单例：

final eventBus = EventBus();

或者创建一个专门的 EventBus 管理类：

class EventBusManager {
  static final EventBusManager _instance = EventBusManager._internal();
  factory EventBusManager() => _instance;

  final EventBus _eventBus = EventBus();

  EventBus get bus => _eventBus;

  EventBusManager._internal();
}

// 全局使用
final eventBus = EventBusManager().bus;

4. 订阅事件

class UserProfilePage extends StatefulWidget {
  @override
  _UserProfilePageState createState() => _UserProfilePageState();
}

class _UserProfilePageState extends State<UserProfilePage> {
  late StreamSubscription _loginSubscription;
  late StreamSubscription _logoutSubscription;

  @override
  void initState() {
    super.initState();

    // 订阅登录事件
    _loginSubscription = eventBus.on<UserLoginEvent>().listen((event) {
      print('用户登录: ${event.username}, Token: ${event.token}');
      // 更新UI，显示用户信息
      setState(() {
        // 更新用户状态
      });
    });

    // 订阅登出事件
    _logoutSubscription = eventBus.on<UserLogoutEvent>().listen((event) {
      print('用户已登出');
      // 清除用户数据，跳转登录页
      Navigator.of(context).pushReplacementNamed('/login');
    });
  }

  @override
  void dispose() {
    // 重要：取消订阅，避免内存泄漏
    _loginSubscription.cancel();
    _logoutSubscription.cancel();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('用户中心')),
      body: Center(child: Text('用户信息')),
    );
  }
}

5. 发布事件

class LoginService {
  Future<void> login(String username, String password) async {
    // 模拟登录请求
    try {
      // ... 登录逻辑

      // 登录成功后发布事件
      eventBus.fire(UserLoginEvent(username, 'mock_token_12345'));

      // 或者发布其他类型的事件
      // eventBus.fire(MessageReceivedEvent('欢迎回来！', DateTime.now()));
    } catch (e) {
      print('登录失败: $e');
    }
  }

  Future<void> logout() async {
    // 模拟登出逻辑
    // ...

    // 发布登出事件
    eventBus.fire(UserLogoutEvent());
  }
}

EventBus 的高级用法

1. 带参数的事件

// 定义复杂的事件类
class OrderEvent {
  final String orderId;
  final double amount;
  final List<String> items;
  final DateTime createTime;

  OrderEvent({
    required this.orderId,
    required this.amount,
    required this.items,
    required this.createTime,
  });
}

// 使用事件
eventBus.fire(OrderEvent(
  orderId: 'ORD20260303',
  amount: 99.99,
  items: ['商品A', '商品B'],
  createTime: DateTime.now(),
));

2. 事件过滤与转换

// 监听所有事件，进行过滤处理
eventBus.on().listen((event) {
  if (event is UserLoginEvent) {
    // 处理登录事件
    AnalyticsService.track('user_login', {'username': event.username});
  } else if (event is OrderEvent) {
    // 处理订单事件
    print('新订单: ${event.orderId}');
  }
});

// 使用 where 进行过滤
eventBus.on<UserLoginEvent>()
  .where((event) => event.username.isNotEmpty)
  .listen((event) {
    print('过滤后的登录事件: ${event.username}');
  });

3. 错误处理

try {
  // 监听事件
  eventBus.on<UserLoginEvent>().listen((event) {
    // 处理事件
  });
} catch (e) {
  print('事件监听错误: $e');
}

4. 条件触发事件

// 在某个条件下触发事件
void processOrder(Order order) async {
  if (order.status == OrderStatus.completed) {
    eventBus.fire(OrderCompletedEvent(order));
  } else if (order.status == OrderStatus.cancelled) {
    eventBus.fire(OrderCancelledEvent(order));
  }
}

实际应用场景

1. 用户登录状态同步

// 登录页面
class LoginPage extends StatelessWidget {
  void _handleLogin(BuildContext context) async {
    await LoginService().login('user', 'pass');
    // 登录成功后，其他页面会自动收到通知
  }
}

// 首页
class HomePage extends StatefulWidget {
  @override
  _HomePageState createState() => _HomePageState();
}

class _HomePageState extends State<HomePage> {
  bool _isLoggedIn = false;

  @override
  void initState() {
    super.initState();
    eventBus.on<UserLoginEvent>().listen((event) {
      setState(() => _isLoggedIn = true);
    });
    eventBus.on<UserLogoutEvent>().listen((event) {
      setState(() => _isLoggedIn = false);
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: _isLoggedIn ? Text('已登录') : Text('未登录'),
    );
  }
}

    );
  }
}

3. 主题切换

class ThemeChangeEvent {
  final ThemeData theme;
  ThemeChangeEvent(this.theme);
}

// 设置页面
class SettingsPage extends StatelessWidget {
  void _changeTheme(BuildContext context, ThemeData theme) {
    eventBus.fire(ThemeChangeEvent(theme));
  }
}

// 应用根组件
class App extends StatefulWidget {
  @override
  _AppState createState() => _AppState();
}

class _AppState extends State<App> {
  ThemeData _currentTheme = ThemeData.light();

  @override
  void initState() {
    super.initState();
    eventBus.on<ThemeChangeEvent>().listen((event) {
      setState(() => _currentTheme = event.theme);
    });
  }

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      theme: _currentTheme,
      home: HomePage(),
    );
  }
}

注意事项

1. 及时取消订阅

@override
void dispose() {
  // 必须取消订阅，防止内存泄漏
  _subscription.cancel();
  super.dispose();
}

2. 避免订阅过多事件

// 不好：订阅所有事件
eventBus.on().listen((event) { ... });

// 好：只订阅需要的特定事件
eventBus.on<SpecificEvent>().listen((event) { ... });

3. 事件命名规范

// 使用清晰的事件命名
class UserLoggedInEvent {}  // ✓
class Login {}              // ✗

class OrderCreatedEvent {}  // ✓
class Create {}             // ✗

4. 考虑使用其他方案

对于复杂的状态管理，建议使用 Provider、Bloc 或 Riverpod 等专业状态管理方案，EventBus 适用于简单的跨组件通信场景。

总结

EventBus 是 Flutter 中实现组件间通信的简单有效的方式。通过发布/订阅模式，我们可以实现松耦合的代码结构，让应用各部分之间的通信更加灵活。在实际开发中，我们应该根据具体场景选择合适的方案：简单的跨组件通信使用 EventBus，复杂的状态管理使用专业的状态管理库。
